{{- /* head.schema.html — JSON-LD structured data */ -}} {{- $schemas := slice -}} {{/* ── WebSite schema (homepage only) ── */}} {{- if .IsHome -}} {{- $ws := dict "@context" "https://schema.org" "@type" "WebSite" "name" .Site.Title "url" .Site.BaseURL -}} {{- with .Site.Params.description -}} {{- $ws = dict "description" . | merge $ws -}} {{- end -}} {{- $pub := dict "@type" "Organization" "name" (or .Site.Params.org_name .Site.Title) "url" .Site.BaseURL -}} {{- $ws = dict "publisher" $pub | merge $ws -}} {{- $schemas = $schemas | append $ws -}} {{- end -}} {{/* ── Article schema (single pages only) ── */}} {{- if and (not .IsHome) (not .IsNode) -}} {{- $article := dict "@context" "https://schema.org" "@type" "Article" "headline" .Title "mainEntityOfPage" (dict "@type" "WebPage" "@id" (.Params.canonical | default .Permalink)) -}} {{- with .Params.description -}} {{- $article = dict "description" (truncate 200 .) | merge $article -}} {{- end -}} {{- with .Params.image -}} {{- $article = dict "image" . | merge $article -}} {{- end -}} {{- $authorName := .Params.author | default "" -}} {{- with $authorName -}} {{- $article = dict "author" (dict "@type" "Person" "name" .) | merge $article -}} {{- end -}} {{- $pub := dict "@type" "Organization" "name" (or .Site.Params.org_name .Site.Title) -}} {{- with .Site.Params.logo -}} {{- $pub = dict "logo" (dict "@type" "ImageObject" "url" .) | merge $pub -}} {{- end -}} {{- $article = dict "publisher" $pub | merge $article -}} {{- with .Date -}} {{- $article = dict "datePublished" (.Format "2006-01-02T15:04:05-07:00") | merge $article -}} {{- end -}} {{- $modDate := .Params.updated | default .Lastmod -}} {{- with $modDate -}} {{- if reflect.IsMap . -}} {{- $article = dict "dateModified" (.Format "2006-01-02T15:04:05-07:00") | merge $article -}} {{- else -}} {{- $article = dict "dateModified" (printf "%s" .) | merge $article -}} {{- end -}} {{- end -}} {{- $schemas = $schemas | append $article -}} {{- end -}} {{/* ── BreadcrumbList schema (all pages except home) ── */}} {{- if not .IsHome -}} {{- $items := slice -}} {{- $items = $items | append (dict "@type" "ListItem" "position" 1 "name" "Home" "item" .Site.BaseURL ) -}} {{- $sections := .CurrentSection -}} {{- if and (not .IsSection) (ne .CurrentSection.RelPermalink "/") -}} {{- $items = $items | append (dict "@type" "ListItem" "position" 2 "name" $sections.Title "item" $sections.Permalink ) -}} {{- end -}} {{- $pos := len $items -}} {{- $pos = add $pos 1 -}} {{- $items = $items | append (dict "@type" "ListItem" "position" $pos "name" .Title "item" (.Params.canonical | default .Permalink) ) -}} {{- $bc := dict "@context" "https://schema.org" "@type" "BreadcrumbList" "itemListElement" $items -}} {{- $schemas = $schemas | append $bc -}} {{- end -}} {{/* ── FAQPage schema (if faq shortcodes were used on this page) ── */}} {{- with .Store.Get "faqs" -}} {{- if gt (len .) 0 -}} {{- $faqItems := slice -}} {{- range $i, $faq := . -}} {{- $faqItems = $faqItems | append (dict "@type" "Question" "name" $faq.q "acceptedAnswer" (dict "@type" "Answer" "text" $faq.a) ) -}} {{- end -}} {{- $faqSchema := dict "@context" "https://schema.org" "@type" "FAQPage" "mainEntity" $faqItems -}} {{- $schemas = $schemas | append $faqSchema -}} {{- end -}} {{- end -}} {{/* ── Output combined JSON-LD ── */}} {{- if gt (len $schemas) 0 -}} {{- end -}}
